            background-color: #f44336;

<body>

    <div class="container">

        <h1>Agree or Disagree?</h1>

        <p id="statement">Loading...</p>

        <button id="agree" onclick="vote('agree')">Agree</button>

        <button id="disagree" onclick="vote('disagree')">Disagree</button>

        <p>Agree: <span id="agreeCount">0</span> | Disagree: <span id="disagreeCount">0</span></p>

    </div>


    <script>

        const statements = [

            "Pineapple belongs on pizza.",

            "Cats are better than dogs.",

            "Technology makes life easier.",

            "Books are better than movies.",

            "Everyone should learn to code."

        ];


        let currentStatementIndex = 0;

        let agreeCount = 0;

        let disagreeCount = 0;


        function loadStatement() {

            document.getElementById("statement").innerText = statements[currentStatementIndex];

        }


        function vote(choice) {

            if (choice === 'agree') {

                agreeCount++;

                document.getElementById("agreeCount").innerText = agreeCount;

            } else {

                disagreeCount++;

                document.getElementById("disagreeCount").innerText = disagreeCount;

            }

            nextStatement();

        }


        function nextStatement() {

            currentStatementIndex++;

            if (currentStatementIndex >= statements.length) {

                currentStatementIndex = 0;

            }

            loadStatement();

        }


        window.onload = loadStatement;

    </script>

</body>
